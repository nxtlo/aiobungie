  83@typing.final
  84class Image:
  85    """A stream-able Bungie resource.
  86
  87    Example
  88    -------
  89    ```py
  90    from aiobungie import Image
  91    img = Image("img/destiny_content/pgcr/raid_eclipse.jpg")
  92    print(img)
  93    # https://www.bungie.net/img/destiny_content/pgcr/raid_eclipse.jpg
  94
  95    # Save the image to a file.
  96    async with img:
  97        await img.save(
  98            "file_name",
  99            "/my/path/to/save/to",
 100            mime_type="png"
 101        )
 102    ```
 103
 104    Parameters
 105    ----------
 106    path : `str | None`
 107        A valid Bungie resource path. if left `None`, `Image.DEFAULT_PATH` will be used.
 108    """
 109
 110    __slots__ = ("path", "_client_session", "_call")
 111    DEFAULT_PATH: typing.ClassVar[str] = "/img/misc/missing_icon_d2.png"
 112    """Returns the path to the missing Bungie image.
 113
 114    This returns the path only, If you want an actual image object use `Image.default()`
 115    """
 116
 117    def __init__(self, path: str | None = None) -> None:
 118        self.path = self.DEFAULT_PATH if not path else path
 119        self._client_session: aiohttp.ClientSession = NotImplemented
 120        self._call: aiohttp.ClientResponse = NotImplemented
 121
 122    @property
 123    def is_missing(self) -> bool:
 124        """Returns `True` if the path of this image is missing.
 125
 126        Example
 127        -------
 128        ```py
 129        # default returns a missing destiny 2 icon.
 130        image = Image.default()
 131        assert image.is_missing
 132
 133        # applies to empty strings as well
 134        image = Image("")
 135        assert image.is_missing
 136        ```
 137        """
 138        return not self.path or self.path == self.DEFAULT_PATH
 139
 140    @staticmethod
 141    def default() -> Image:
 142        """Return the default image.
 143
 144        Some Bungie resources can be nullable, aiobungie usually replaces them with this instance.
 145        but not always, they might be `Image | None` in certain cases.
 146
 147        Example
 148        -------
 149        ```py
 150        missing = Image.default()
 151        print(missing.create_url()) # https://www.bungie.net/img/misc/missing_icon_d2.png
 152        ```
 153        """
 154        return _DEFAULT_IMAGE
 155
 156    def create_url(self) -> str:
 157        """Creates a full URL to the image path.
 158
 159        Example
 160        -------
 161        ```py
 162        img = Image("img/destiny_content/pgcr/raid_eclipse.jpg")
 163        print(img.create_url())
 164        # https://www.bungie.net/img/destiny_content/pgcr/raid_eclipse.jpg
 165        ```
 166
 167        Returns
 168        -------
 169        `str`
 170            The URL to the image.
 171        """
 172        return url.BASE + "/" + self.path
 173
 174    async def static_request(self) -> aiohttp.ClientResponse:
 175        client_session = aiohttp.ClientSession()
 176        request = client_session.request(
 177            "GET", self.create_url(), raise_for_status=False
 178        )
 179        try:
 180            await client_session.__aenter__()
 181            response = await request.__aenter__()
 182            try:
 183                if 300 >= response.status >= 200:
 184                    self._client_session = client_session
 185                    self._call = response
 186                    return response
 187                else:
 188                    raise await error.panic(response)
 189
 190            except Exception as e:
 191                await response.__aexit__(type(e), e, e.__traceback__)
 192                raise
 193
 194        except Exception:
 195            await client_session.close()
 196            raise
 197
 198    async def save(
 199        self,
 200        file_name: str,
 201        path: str | os.PathLike[str],
 202        *,
 203        mime_type: MimeType | str = MimeType.JPEG,
 204        executor: concurrent.futures.Executor | None = None,
 205    ) -> None:
 206        """Saves this image to a file.
 207
 208        Parameters
 209        ----------
 210        file_name : `str`
 211            A name for the file to save the image to.
 212        path : `PathLike[str] | str`
 213            A path to save the image to.
 214
 215        Other Parameters
 216        ----------------
 217        mime_type : `MimeType | str`
 218            MIME type of the image. Defaults to JPEG.
 219        executor : `concurrent.futures.Executor | None`
 220            An optional executor to use for writing the bytes of this image.
 221
 222        Raises
 223        ------
 224        `FileNotFoundError`
 225            If the path provided does not exist.
 226        `PermissionError`
 227            If the path provided is not writable or does not have write permissions.
 228        `RuntimeError`
 229            If the image could not be saved for some other reason.
 230        """
 231        if isinstance(path, pathlib.Path) and not path.exists():
 232            raise FileNotFoundError(f"File does not exist: {path!r}")
 233
 234        # empty str, nothing to do here.
 235        if self.is_missing and self.path != self.DEFAULT_PATH:
 236            return
 237
 238        path = pathlib.Path(path) / f"{file_name}.{mime_type}"
 239        loop = helpers.get_or_make_loop()
 240        file = await loop.run_in_executor(executor, _open_write_path, path)
 241
 242        reader = await self.static_request()
 243        try:
 244            async for chunk in reader.content:
 245                await loop.run_in_executor(executor, file.write, chunk)
 246
 247        except asyncio.CancelledError:
 248            pass
 249
 250        except Exception as err:
 251            raise RuntimeError("Encountered an error while saving image.") from err
 252
 253    async def read(self) -> bytes:
 254        """Perform an HTTP call reading this image's entire bytes into memory.
 255
 256        Example
 257        -------
 258        ```py
 259        image = Image.default()
 260        # you can fetch the bytes in two different ways
 261        # they do the exact same thing.
 262        async with image:
 263            buffer = await image.read() or await image
 264        ```
 265
 266        Returns
 267        -------
 268        `bytes`:
 269            The bytes of this image.
 270        """
 271        return await (await self.static_request()).read()
 272
 273    async def stream(self) -> aiohttp.streams.AsyncStreamIterator[bytes]:
 274        """Stream this image's data.
 275
 276        Example
 277        -------
 278        ```py
 279        image = Image.default()
 280        async with image:
 281            stream = await image.stream()
 282            async for byte in stream:
 283                # write chunk to file
 284            ...
 285        ```
 286
 287        Returns
 288        -------
 289        `AsyncStreamIterator[bytes]`:
 290            A lazy async iterator that is ready in memory.
 291        """
 292        return (await self.static_request()).content.iter_any()
 293
 294    async def chunks(self, size: int) -> aiohttp.streams.AsyncStreamIterator[bytes]:
 295        """Stream this image's data in chunks.
 296
 297        Example
 298        -------
 299        ```py
 300        # it must be awaited to fetch the image first.
 301        buffer_size = 1024
 302        image = Image.default()
 303
 304        async with image:
 305            chunks = await image.chunks(buffer_size)
 306            while True:
 307                next_chunk = await anext(chunks)
 308                if not next_chunk:
 309                    break
 310                # write chunk to file
 311        ```
 312
 313        Returns
 314        -------
 315        `AsyncStreamIterator[bytes]`:
 316            lazy async iterator that is ready in memory.
 317        """
 318        return (await self.static_request()).content.iter_chunked(size)
 319
 320    async def iter(self) -> collections.AsyncGenerator[bytes, None]:
 321        """Yield each byte in this image from start to end.
 322
 323        Example
 324        -------
 325        ```py
 326        resource = Image.default()
 327        async with resource:
 328            async for byte in resource.iter():
 329                print(byte)
 330        ```
 331
 332        Returns
 333        -------
 334        `collections.AsyncGenerator[bytes, None]`
 335            An async generator that yields this image's bytes.
 336        """
 337
 338        reader = await self.chunks(1024)
 339        async for chunk in reader:
 340            yield chunk
 341
 342    async def __aenter__(self) -> Self:
 343        return self
 344
 345    async def __aexit__(
 346        self,

 355        if isinstance(other, str):
 356            return self.path == other

 360        return NotImplemented

 372        return self.create_url()

 423@attrs.frozen(kw_only=True, slots=False)
 424class OAuthURL:
 425    """The result of calling `aiobungie.RESTClient.build_oauth2_url`.
 426
 427    Example
 428    -------
 429    ```py
 430    url = aiobungie.builders.OAuthURL(client_id=1234)
 431    print(url.compile()) # Full URL to make an OAuth2 request.
 432    print(url.state)  # The UUID state to be used in the OAuth2 request.
 433    ```
 434    """
 435
 436    state: uuid.UUID = attrs.field(factory=uuid.uuid4)
 437    """The state parameter for the URL."""
 438
 439    client_id: int
 440    """The client id for that's making the request."""
 441
 442    @functools.cached_property
 443    def url(self) -> str:
 444        """The cached result of calling `OAuthURL.compile`.
 445
 446        This allow for a faster access for the compiled URL.
 447
 448        Example
 449        -------
 450        ```py
 451        url = OAuthURL(client_id=0000)
 452        first_compile = url.url
 453        cached_url = url.url
 454        ```
 455        """
 456        return self.compile()
 457
 458    def compile(self) -> str:
 459        """Compiles the fields to finalize the result of the URL."""
 460        return (
 461            url.OAUTH_EP
 462            + f"?client_id={self.client_id}&response_type=code&state={self.state}"  # noqa: W503
 463        )
 464
 465    def __str__(self) -> str:
 466        return self.compile()
