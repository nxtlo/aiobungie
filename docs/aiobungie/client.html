  59class Client(traits.ClientApp):
  60    """Standard Bungie API client application.
  61
  62    This client deserialize the REST JSON responses using `aiobungie.internal.factory.Factory`
  63    and returns `aiobungie.crates` Python object implementations of the responses.
  64
  65    A `aiobungie.RESTClient` REST client can also be used alone for low-level concepts.
  66
  67    Example
  68    -------
  69    ```py
  70    import aiobungie
  71
  72    client = aiobungie.Client('...')
  73
  74    async def main():
  75        async with client.rest:
  76            user = await client.fetch_current_user_memberships('...')
  77            print(user)
  78    ```
  79
  80    Parameters
  81    -----------
  82    token: `str`
  83        Your Bungie's API key or Token from the developer's portal.
  84
  85    Other Parameters
  86    ----------------
  87    max_retries : `int`
  88        The max retries number to retry if the request hit a `5xx` status code.
  89    client_secret : `str | None`
  90        An optional application client secret,
  91        This is only needed if you're fetching OAuth2 tokens with this client.
  92    client_id : `int | None`
  93        An optional application client id,
  94        This is only needed if you're fetching OAuth2 tokens with this client.
  95    """
  96

  99    def __init__(
 100        self,
 101        token: str,
 102        /,
 103        *,
 104        client_secret: typing.Optional[str] = None,
 105        client_id: typing.Optional[int] = None,
 106        max_retries: int = 4,
 107    ) -> None:
 108        self._rest = rest_.RESTClient(
 109            token,
 110            client_secret=client_secret,
 111            client_id=client_id,
 112            max_retries=max_retries,
 113        )
 114
 115        self._factory = factory_.Factory(self)

 133    def run(
 134        self, future: collections.Coroutine[typing.Any, None, None], debug: bool = False
 135    ) -> None:
 136        loop: typing.Final[asyncio.AbstractEventLoop] = helpers.get_or_make_loop()
 137        try:
 138            if not loop.is_running():
 139                loop.set_debug(debug)
 140                loop.run_until_complete(future)
 141
 142        except Exception as exc:
 143            raise RuntimeError(f"Failed to run {future.__qualname__}") from exc
 144
 145        except KeyboardInterrupt:
 146            _LOG.warn("Unexpected Keyboard interrupt. Exiting.")
 147            return
