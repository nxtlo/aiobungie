106        if self.message:

113            f"{self.http_status.name.replace('_', '').title()} {self.http_status.value}: "
114            f"Error status: {error_status_body}, Error message: {message_body} from {self.url} "
115            f"{str(self.body)}"

228async def raise_error(response: aiohttp.ClientResponse) -> AiobungieError:
229    """Generates and raise exceptions on error responses."""
230
231    # Not a JSON response, raise immediately.
232
233    # Also Bungie sometimes get funky and return HTML instead of JSON when making an authorized
234    # request with a dummy access token. I can't really do anything about this..
235    if response.content_type != "application/json":
236        return HTTPError(
237            f"Expected JSON content but got {response.content_type!s}, {response.real_url!s}",
238            http.HTTPStatus.UNSUPPORTED_MEDIA_TYPE,
239        )
240
241    body = await response.json()
242    message: str = body.get("Message", "UNDEFINED_MESSAGE")
243    error_status: str = body.get("ErrorStatus", "UNDEFINED_ERROR_STATUS")
244    message_data: dict[str, str] = body.get("MessageData", {})
245    throttle_seconds: int = body.get("ThrottleSeconds", 0)
246    error_code: int = body.get("ErrorCode", 0)
247
248    # Standard HTTP status.
249    if response.status == http.HTTPStatus.NOT_FOUND:
250        return NotFound(
251            message=message,
252            error_code=error_code,
253            throttle_seconds=throttle_seconds,
254            url=str(response.real_url),
255            body=body,
256            headers=response.headers,
257            error_status=error_status,
258            message_data=message_data,
259        )
260
261    elif response.status == http.HTTPStatus.FORBIDDEN:
262        return Forbidden(
263            message=message,
264            error_code=error_code,
265            throttle_seconds=throttle_seconds,
266            url=str(response.real_url),
267            body=body,
268            headers=response.headers,
269            error_status=error_status,
270            message_data=message_data,
271        )
272
273    elif response.status == http.HTTPStatus.UNAUTHORIZED:
274        return Unauthorized(
275            message=message,
276            error_code=error_code,
277            throttle_seconds=throttle_seconds,
278            url=str(response.real_url),
279            body=body,
280            headers=response.headers,
281            error_status=error_status,
282            message_data=message_data,
283        )
284
285    elif response.status == http.HTTPStatus.BAD_REQUEST:
286        # Membership needs to be alone.
287        if error_status == "InvalidParameters":
288            return MembershipTypeError(
289                message=message,
290                body=body,
291                headers=response.headers,
292                url=str(response.url),
293                membership_type=message_data["membershipType"],
294                required_membership=message_data["membershipInfo.membershipType"],
295                membership_id=int(message_data["membershipId"]),
296            )
297        return BadRequest(
298            message=message,
299            body=body,
300            headers=response.headers,
301            url=str(response.url),
302        )
303
304    status = http.HTTPStatus(response.status)
305
306    if 400 <= status < 500:
307        return ResponseError(
308            message=message,
309            error_code=error_code,
310            throttle_seconds=throttle_seconds,
311            url=str(response.real_url),
312            body=body,
313            headers=response.headers,
314            error_status=error_status,
315            message_data=message_data,
316            http_status=status,
317        )
318
319    # Need to self handle ~5xx errors
320    elif 500 <= status < 600:
321        # No API key or method requires OAuth2 most likely.
322        if error_status in {
323            "ApiKeyMissingFromRequest",
324            "WebAuthRequired",
325            "ApiInvalidOrExpiredKey",
326            "AuthenticationInvalid",
327            "AuthorizationCodeInvalid",
328        }:
329            return Unauthorized(
330                message=message,
331                error_code=error_code,
332                throttle_seconds=throttle_seconds,
333                url=str(response.real_url),
334                body=body,
335                headers=response.headers,
336                error_status=error_status,
337                message_data=message_data,
338            )
339
340        # Anything contains not found.
341        elif (
342            "NotFound" in error_status or error_status == "UserCannotFindRequestedUser"
343        ):
344            return NotFound(
345                message=message,
346                error_code=error_code,
347                throttle_seconds=throttle_seconds,
348                url=str(response.real_url),
349                body=body,
350                headers=response.headers,
351                error_status=error_status,
352                message_data=message_data,
353            )
354
355        # Other 5xx errors.
356        else:
357            return InternalServerError(
358                message=message,
359                error_code=error_code,
360                throttle_seconds=throttle_seconds,
361                url=str(response.real_url),
362                body=body,
363                headers=response.headers,
364                error_status=error_status,
365                message_data=message_data,
366                http_status=status,
367            )
368    # Something else.
369    else:
370        return HTTPException(
371            message=message,
372            error_code=error_code,
373            throttle_seconds=throttle_seconds,
374            url=str(response.real_url),
375            body=body,
376            headers=response.headers,
377            error_status=error_status,
378            message_data=message_data,
379            http_status=status,
380        )

383def stringify_http_message(headers: collections.Mapping[str, str]) -> str:
384    return (
385        "{ \n"
386        + "\n".join(  # noqa: W503
387            f"{f'   {key}'}: {value}"
388            if key not in ("Authorization", "X-API-KEY")
389            else f"   {key}: HIDDEN_TOKEN"
390            for key, value in headers.items()
391        )
392        + "\n}"  # noqa: W503
393    )

229async def raise_error(response: aiohttp.ClientResponse) -> AiobungieError:
230    """Generates and raise exceptions on error responses."""
231
232    # Not a JSON response, raise immediately.
233
234    # Also Bungie sometimes get funky and return HTML instead of JSON when making an authorized
235    # request with a dummy access token. I can't really do anything about this..
236    if response.content_type != "application/json":
237        return HTTPError(
238            f"Expected JSON content but got {response.content_type!s}, {response.real_url!s}",
239            http.HTTPStatus.UNSUPPORTED_MEDIA_TYPE,
240        )
241
242    body = await response.json()
243    message: str = body.get("Message", "UNDEFINED_MESSAGE")
244    error_status: str = body.get("ErrorStatus", "UNDEFINED_ERROR_STATUS")
245    message_data: dict[str, str] = body.get("MessageData", {})
246    throttle_seconds: int = body.get("ThrottleSeconds", 0)
247    error_code: int = body.get("ErrorCode", 0)
248
249    # Standard HTTP status.
250    if response.status == http.HTTPStatus.NOT_FOUND:
251        return NotFound(
252            message=message,
253            error_code=error_code,
254            throttle_seconds=throttle_seconds,
255            url=str(response.real_url),
256            body=body,
257            headers=response.headers,
258            error_status=error_status,
259            message_data=message_data,
260        )
261
262    elif response.status == http.HTTPStatus.FORBIDDEN:
263        return Forbidden(
264            message=message,
265            error_code=error_code,
266            throttle_seconds=throttle_seconds,
267            url=str(response.real_url),
268            body=body,
269            headers=response.headers,
270            error_status=error_status,
271            message_data=message_data,
272        )
273
274    elif response.status == http.HTTPStatus.UNAUTHORIZED:
275        return Unauthorized(
276            message=message,
277            error_code=error_code,
278            throttle_seconds=throttle_seconds,
279            url=str(response.real_url),
280            body=body,
281            headers=response.headers,
282            error_status=error_status,
283            message_data=message_data,
284        )
285
286    elif response.status == http.HTTPStatus.BAD_REQUEST:
287        # Membership needs to be alone.
288        if error_status == "InvalidParameters":
289            return MembershipTypeError(
290                message=message,
291                body=body,
292                headers=response.headers,
293                url=str(response.url),
294                membership_type=message_data["membershipType"],
295                required_membership=message_data["membershipInfo.membershipType"],
296                membership_id=int(message_data["membershipId"]),
297            )
298        return BadRequest(
299            message=message,
300            body=body,
301            headers=response.headers,
302            url=str(response.url),
303        )
304
305    status = http.HTTPStatus(response.status)
306
307    if 400 <= status < 500:
308        return ResponseError(
309            message=message,
310            error_code=error_code,
311            throttle_seconds=throttle_seconds,
312            url=str(response.real_url),
313            body=body,
314            headers=response.headers,
315            error_status=error_status,
316            message_data=message_data,
317            http_status=status,
318        )
319
320    # Need to self handle ~5xx errors
321    elif 500 <= status < 600:
322        # No API key or method requires OAuth2 most likely.
323        if error_status in {
324            "ApiKeyMissingFromRequest",
325            "WebAuthRequired",
326            "ApiInvalidOrExpiredKey",
327            "AuthenticationInvalid",
328            "AuthorizationCodeInvalid",
329        }:
330            return Unauthorized(
331                message=message,
332                error_code=error_code,
333                throttle_seconds=throttle_seconds,
334                url=str(response.real_url),
335                body=body,
336                headers=response.headers,
337                error_status=error_status,
338                message_data=message_data,
339            )
340
341        # Anything contains not found.
342        elif (
343            "NotFound" in error_status or error_status == "UserCannotFindRequestedUser"
344        ):
345            return NotFound(
346                message=message,
347                error_code=error_code,
348                throttle_seconds=throttle_seconds,
349                url=str(response.real_url),
350                body=body,
351                headers=response.headers,
352                error_status=error_status,
353                message_data=message_data,
354            )
355
356        # Other 5xx errors.
357        else:
358            return InternalServerError(
359                message=message,
360                error_code=error_code,
361                throttle_seconds=throttle_seconds,
362                url=str(response.real_url),
363                body=body,
364                headers=response.headers,
365                error_status=error_status,
366                message_data=message_data,
367                http_status=status,
368            )
369    # Something else.
370    else:
371        return HTTPException(
372            message=message,
373            error_code=error_code,
374            throttle_seconds=throttle_seconds,
375            url=str(response.real_url),
376            body=body,
377            headers=response.headers,
378            error_status=error_status,
379            message_data=message_data,
380            http_status=status,
381        )
