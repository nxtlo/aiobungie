 70class Image:
 71    """Representation of an image/avatar/picture at Bungie.
 72
 73    Example
 74    -------
 75    ```py
 76    from aiobungie import Image
 77    img = Image("img/destiny_content/pgcr/raid_eclipse.jpg")
 78    print(img)
 79    # https://www.bungie.net/img/destiny_content/pgcr/raid_eclipse.jpg
 80
 81    # Stream the image.
 82    async for chunk in img:
 83        # Byte chunks of the image.
 84        print(chunk)
 85
 86    # Save the image to a file.
 87    await img.save("file_name", "/my/path/to/save/to", "jpeg")
 88    ```
 89
 90    Parameters
 91    ----------
 92    path : `str | None`
 93        The path to the image. If `None`, the default missing image path will be used.
 94    """
 95
 96    __slots__ = ("_path",)
 97
 98    def __init__(self, path: typing.Optional[str] = None) -> None:
 99        self._path = path
100
101    @property
102    def is_missing(self) -> bool:
103        return not self._path
104
105    @property
106    def url(self) -> str:
107        """The URL to the image."""
108        return self.create_url()
109
110    @staticmethod
111    def missing_path() -> str:
112        """Returns the path to the missing Bungie image."""
113        return "img/misc/missing_icon_d2.png"
114
115    def create_url(self) -> str:
116        """Creates a full URL to the image path.
117
118        Returns
119        -------
120        str
121            The URL to the image.
122        """
123        return f"{url.BASE}/{self._path if self._path else self.missing_path()}"
124
125    async def save(
126        self,
127        file_name: str,
128        path: typing.Union[pathlib.Path, str],
129        /,
130        mime_type: typing.Optional[typing.Union[MimeType, str]] = None,
131    ) -> None:
132        """Saves the image to a file.
133
134        Parameters
135        ----------
136        file_name : `str`
137            A name for the file to save the image to.
138        path : `pathlib.Path | str`
139            A path tp save the image to.
140
141        Other Parameters
142        ----------------
143        mime_type : `MimeType | str`
144            Optional MIME type of the image.
145
146        Raises
147        ------
148        `FileNotFoundError`
149            If the path provided does not exist.
150        `RuntimeError`
151            If the image could not be saved.
152        `PermissionError`
153            If the path provided is not writable or does not have write permissions.
154        """
155        if isinstance(path, pathlib.Path) and not path.exists():
156            raise FileNotFoundError(f"File does not exist: {path!r}")
157
158        if self.is_missing:
159            return
160
161        mimetype = mime_type or MimeType.PNG
162        path = pathlib.Path(path)
163
164        loop = helpers.get_or_make_loop()
165        pool = concurrent.futures.ThreadPoolExecutor()
166
167        try:
168            with pool:
169                await loop.run_in_executor(
170                    pool, _write, path, file_name, mimetype, await self.read()
171                )
172                _LOGGER.info("Saved image to %s", file_name)
173
174        except asyncio.CancelledError:
175            pass
176
177        except Exception as err:
178            raise RuntimeError("Encountered an error while saving image.") from err
179
180    async def read(self) -> bytes:
181        """Read this image bytes.
182
183        Returns
184        -------
185        `bytes`
186            The bytes of this image.
187        """
188        client_session = aiohttp.ClientSession()
189
190        try:
191            await client_session.__aenter__()
192            response = await client_session.get(self.create_url())
193
194            if 300 >= response.status >= 200:
195                reader = await response.read()
196
197        except Exception as exc:
198            raise RuntimeError(f"Failed to read image: {exc}") from None
199        finally:
200            await client_session.__aexit__(None, None, None)
201        return reader
202
203    async def iter(self) -> collections.AsyncGenerator[bytes, None]:
204        """Iterates over the image bytes lazily.
205
206        Example
207        -------
208        import aiobungie
209
210        resource = aiobungie.Image("img/misc/missing_icon_d2.png")
211        async for chunk in resource.iter():
212            print(chunk)
213
214        Returns
215        -------
216        `collections.AsyncGenerator[bytes, None]`
217            An async generator of the image bytes.
218        """
219
220        async for chunk in self:
221            yield chunk
222
223    def __repr__(self) -> str:
224        return f"Image(url={self.create_url()})"
225
226    def __str__(self) -> str:
227        return self.create_url()
228
229    def __aiter__(self) -> Image:
230        return self
231
232    async def __anext__(self) -> bytes:
233        return await self.read()
234
235    def __await__(self) -> collections.Generator[None, None, bytes]:
236        return self.__anext__().__await__()
