import enum as __enum
import typing


class IntEnum(__enum.IntEnum):
    """An int only enum."""

    def __int__(self) -> int:
        return int(self.value)

    def __str__(self) -> str:
        return self.name


class Enum(__enum.Enum):
    """An enum that can be an int or a string."""

    def __str__(self) -> str:
        return self.name

    def __int__(self) -> int:
        return int(self.value)
