 44class Iterator(typing.Generic[Item]):
 45    """A Flat, In-Memory iterator for sequenced based data.
 46
 47    Example
 48    -------
 49    ```py
 50    iterator = Iterator([1, 2, 3])
 51
 52    # Map the results.
 53    for item in iterator.map(lambda item: item * 2):
 54        print(item)
 55    # 2
 56    # 4
 57
 58    # Indexing is also supported.
 59    print(iterator[0])
 60    # 1
 61
 62    # Normal iteration.
 63    for item in iterator:
 64        print(item)
 65    # 1
 66    # 2
 67    # 3
 68
 69    # Union two iterators.
 70    iterator2 = Iterator([4, 5, 6])
 71    final = iterator | iterator2
 72    # <Iterator([1, 2, 3, 4, 5, 6])>
 73    ```
 74
 75    Parameters
 76    ----------
 77    items: `collections.Iterable[Item]`
 78        The items to iterate over.
 79    """
 80
 81    __slots__ = ("_items",)
 82
 83    def __init__(self, items: collections.Iterable[Item]) -> None:
 84        self._items = iter(items)

 94    def collect(
 95        self, casting: typing.Optional[_B] = None
 96    ) -> typing.Union[list[Item], list[_B]]:
 97        """Collects all items in the iterator into a list and cast them into an object if provided.
 98
 99        Example
100        -------
101        >>> iterator = Iterator([1, 2, 3])
102        >>> iterator.collect(casting=str)
103        ["1", "2", "3"]
104
105        Parameters
106        ----------
107        casting: `T | None`
108            The type to cast the items to. If `None` is provided, the items will be returned as is.
109
110        Raises
111        ------
112        `StopIteration`
113            If no elements are left in the iterator.
114        """
115        if casting is not None:
116            return typing.cast(list[_B], list(map(casting, self._items)))
117
118        return list(self._items)

120    def next(self) -> Item:
121        """Returns the next item in the iterator.
122
123        Example
124        -------
125        ```py
126        iterator = Iterator(["1", "2", "3"])
127        item = iterator.next()
128        assert item == "1"
129        item = iterator.next()
130        assert item == "2"
131        ```
132
133        Raises
134        ------
135        `StopIteration`
136            If no elements are left in the iterator.
137        """
138        try:
139            return self.__next__()
140        except StopIteration:
141            self._ok()

143    def map(
144        self, predicate: collections.Callable[[Item], OtherItem]
145    ) -> Iterator[OtherItem]:
146        """Maps each item in the iterator to its predicated value.
147
148        Example
149        -------
150        ```py
151        iterator = Iterator(["1", "2", "3"]).map(lambda value: int(value))
152        print(iterator)
153        # <Iterator([1, 2, 3])>
154        ```
155
156        Parameters
157        ----------
158        predicate: `collections.Callable[[Item], OtherItem]`
159            The function to map each item in the iterator to its predicated value.
160
161        Returns
162        -------
163        `Iterator[OtherItem]`
164            The mapped iterator.
165
166        Raises
167        ------
168        `StopIteration`
169            If no elements are left in the iterator.
170        """
171        return Iterator(map(predicate, self._items))

173    def take(self, n: int) -> Iterator[Item]:
174        """Take the first number of items until the number of items are yielded or
175        the end of the iterator is reached.
176
177        Example
178        -------
179        ```py
180        iterator = Iterator([GameMode.RAID, GameMode.STRIKE, GameMode.GAMBIT])
181        print(iterator.take(2))
182        # <Iterator([GameMode.RAID, GameMode.STRIKE])>
183        ```
184
185        Parameters
186        ----------
187        n: `int`
188            The number of items to take.
189
190        Raises
191        ------
192        `StopIteration`
193            If no elements are left in the iterator.
194        """
195        return Iterator(itertools.islice(self._items, n))

298    def all(self, predicate: collections.Callable[[Item], bool]) -> bool:
299        """`True` if all items in the iterator match the predicate.
300
301        Example
302        -------
303        ```py
304        iterator = Iterator([1, 2, 3])
305        while iterator.all(lambda item: isinstance(item, int)):
306            print("Still all integers")
307            continue
308        # Still all integers
309        ```

311        Parameters
312        ----------
313        predicate: `collections.Callable[[Item], bool]`

315
316        Raises
317        ------
318        `StopIteration`
319            If no elements are left in the iterator.

321        return all(predicate(item) for item in self)

475    async def async_for_each(
476        self,
