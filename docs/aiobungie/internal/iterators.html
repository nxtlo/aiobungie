 42class FlatIterator(typing.Generic[Item]):
 43    """A Flat, In-Memory iterator for sequenced based data.
 44
 45    This can either be used sync or asynchronously.
 46
 47    Example
 48    -------
 49    ```py
 50    iterator = FlatIterator([1, 2, 3])
 51
 52    # Limit the results to 2.
 53    async for item in iterator.take(2):
 54        print(item)
 55    # 1
 56    # 2
 57
 58    # Filter the results.
 59    async for item in iterator.filter(lambda item: item > 1):
 60        print(item)
 61        print(iterator.count())
 62    # 2
 63    # 3
 64    # 3
 65
 66    # Map the results.
 67    async for item in iterator.map(lambda item: item * 2):
 68        print(item)
 69    # 2
 70    # 4
 71
 72    # This also works synchronously.
 73    iterator = FlatIterator(["Hello", "World", "!"])
 74    for item in iterator.discard(lambda item: "!" in item):
 75        print(item)
 76    # Hello
 77    # World
 78
 79    # Indexing is also supported.
 80
 81    print(iterator[0])
 82    # Hello
 83    ```
 84
 85    Parameters
 86    ----------
 87    items: `collections.Iterable[Item]`
 88        The items to iterate over.
 89    """

122    def collect(
123        self, casting: typing.Optional[_B] = None
124    ) -> typing.Union[list[Item], list[_B]]:
125        """Collects all items in the iterator into a list.
126
127        Example
128        -------
129        >>> iterator = FlatIterator([1, 2, 3])
130        >>> iterator.collect()
131        [1, 2, 3]
132
133        Raises
134        ------
135        `StopIteration`
136            If no elements are left in the iterator.
137        """
138        if casting is not None:
139            return typing.cast(list[_B], list(map(casting, self._items)))
140
141        return list(self._items)

143    def next(self) -> Item:
144        """Returns the next item in the iterator.
145
146        Example
147        -------
148        >>> iterator = FlatIterator[str](["1", "2", "3"])
149        item = iterator.next()
150        assert item == "1"
151        item = iterator.next()
152        assert item == "2"
153
154        Raises
155        ------
156        `StopIteration`
157            If no elements are left in the iterator.
158        """
159        try:
160            return self.__next__()
161        except StopIteration:
162            self._ok()

165        self, predicate: collections.Callable[[Item], OtherItem]
166    ) -> FlatIterator[OtherItem]:
167        """Maps each item in the iterator to its predicated value.

184            If no elements are left in the iterator.

188    def take(self, n: int) -> FlatIterator[Item]:
189        """Take the first number of items until the number of items are yielded or

202            The number of items to take.

206        `StopIteration`
207            If no elements are left in the iterator.

209        return FlatIterator(itertools.islice(self._items, n))

311    def discard(
312        self, predicate: collections.Callable[[Item], bool]
313    ) -> FlatIterator[Item]:
314        """Discards all elements in the iterator for which the predicate function returns true.
315
316        Example
317        -------
318        >>> iterator = FlatIterator([MembershipType.STEAM, MembershipType.XBOX, MembershipType.STADIA])
319        >>> async for _ in iterator.discard(lambda platform: platform is not MembershipType.STEAM):
320                # Drops all memberships that are not steam.
321                print(iterator)

323
324        Parameters
325        ----------
326        predicate: `collections.Callable[[Item], bool]`

328
329        Raises
330        ------

332            If no elements are left in the iterator.
333        """
334        return FlatIterator(filter(lambda x: not predicate(x), self._items))

A Flat, In-Memory iterator for sequenced based data.

This can either be used sync or asynchronously.

Example
iterator = FlatIterator([1, 2, 3])

# Limit the results to 2.
async for item in iterator.take(2):
    print(item)
# 1
# 2

# Filter the results.
async for item in iterator.filter(lambda item: item > 1):
    print(item)
    print(iterator.count())
# 2
# 3
# 3

# Map the results.
async for item in iterator.map(lambda item: item * 2):
    print(item)
# 2
# 4

# This also works synchronously.
iterator = FlatIterator(["Hello", "World", "!"])
for item in iterator.discard(lambda item: "!" in item):
    print(item)
# Hello
# World

# Indexing is also supported.

print(iterator[0])
# Hello
