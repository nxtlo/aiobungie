def deprecated(func):
    """
    functions with this decorator will not work or is not implemented yet.
    """

    @wraps(func)
    def wrapper(*args: typing.Any, **kwargs: typing.Any) -> None:
        warnings.warn(
            f"function {func.__name__!r} deprecated until further notice.",
            stacklevel=2,
            category=DeprecationWarning,
        )
        func.__doc__ += " DEPRECATED FUNCTION."
        coro = func(*args, **kwargs)
        return coro

    return wrapper

class Image:
    def __init__(self, path: str) -> None:
        self.path = path

    def __str__(self) -> str:
        return url.BASE + self.path

    @property
    def url(self) -> str:
        return str(self)

    @property
    def partial(self) -> str:
        if self.path is not None:
            return self.path
        return f"Image<{helpers.Undefined}>"
